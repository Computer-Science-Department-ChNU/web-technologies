      ];
       
      function getCartPrice(array) {
        let price = 0; // загальна вартість (поки — 0)
        for (let i = 0; i < array.length; i = i + 1) {
          price = price + ( array[i].price * array[i].quantity );
        }
        return price; // повернути підраховану вартість
      }
       
      let cartPrice = getCartPrice(cart); // 1200
